var primernumero;
var segundonumero;
var numerouno;
var numerodos;
var suma;

primernumero = window.prompt("Introduce el primer número");
segundonumero = window.prompt("Introduce el segundo número");

numerouno = parseInt(primernumero);
numerodos = parseInt(segundonumero);
suma = numerouno + numerodos;

// Mostrar el resultado en un cuadro de diálogo
window.alert("La suma es: " + suma);
